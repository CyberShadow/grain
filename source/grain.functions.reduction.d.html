8 struct Sum(string mode = "fast", T, size_t dim) {
9     import std.traits : isFloatingPoint;
10     static assert(isFloatingPoint!T, "currently only float point is supported.");
11 
12     uint[dim] shape;
13 
14     auto forward(Variable!(T, dim, HostStorage) x) {
15         import mir.math : sum;
16 
17         // TODO if train
18         this.shape = x.shape;
19         auto result = x.sliced.sum!mode;
20         return result.variable;
21     }
22 
23     auto backward(Variable!(T, 0, HostStorage) y) {
24         auto gx = uninitVariable!T(this.shape, y.requiresGrad);
25         import std.algorithm : fill;
26         fill(gx.data, y.data[0]);
27         return gx;
28     }
